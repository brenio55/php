strtoupper([variável]) -> põe a variável para maiúsculo;

strtolower([variável]) -> põe a variável para minúsculo;

ucwords([variável]) -> põe as primeiras letras das palavras em maiúsculo;

ucfirst([variável]) -> põe só a primeira letra das palavras inteiras em maiúsculo;

str_replace(valor que se quer trocar, valor que se quer adicionar, $variável)

str_pos(variável, "valor requerido") -> retorna a posição na string do texto requerido.

substr(variável, posição, posição final) -> corta string.

strlen(variável) -> retorna o tamanho da variável
